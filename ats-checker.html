// ===== Premium ATS Logic (Frontend Only) ===== function cleanResume(text) { return text .replace(/linkedin\.com\/[^\s]+/gi, "") .replace(/email:.*|phone:.*|contact:.*|address:.*/gi, "") .replace(/\n{2,}/g, "\n"); } function extractSentences(text) { return text .split(/[.!?]/) .map(s => s.trim()) .filter(s => s.length > 20); } function scoreSentence(s) { let score = 0; if (/\d+%?|\d+\+|₹|\$/.test(s)) score += 2; if (/led|built|improved|increased|reduced|designed|launched/i.test(s)) score += 2; if (s.split(" ").length > 12 && s.split(" ").length < 30) score += 1; return score; } async function extract(file) { const ext = file.name.split(".").pop(); if (ext === "txt") return await file.text(); if (ext === "pdf") { const pdf = await pdfjsLib.getDocument({ data: await file.arrayBuffer() }).promise; let text = ""; for (let i = 1; i <= pdf.numPages; i++) { const page = await pdf.getPage(i); const content = await page.getTextContent(); text += content.items.map(x => x.str).join(" "); } return text; } if (ext === "docx") { const result = await mammoth.extractRawText({ arrayBuffer: await file.arrayBuffer() }); return result.value; } } function runPremiumLoading(done) { const loading = document.getElementById("loading"); const results = document.getElementById("results"); const text = loading.querySelector("p"); const steps = [ "Parsing resume content…", "Ignoring headers & contact info…", "Evaluating sentence impact…", "Calculating ATS probability…" ]; loading.classList.remove("hidden"); results.classList.add("hidden"); let i = 0; const timer = setInterval(() => { if (i >= steps.length) { clearInterval(timer); loading.classList.add("hidden"); done(); } else { text.innerText = steps[i]; i++; } }, 800); } async function analyzeATS() { const file = document.getElementById("file").files[0]; if (!file) return alert("Upload a resume first"); runPremiumLoading(async () => { const raw = await extract(file); const resume = cleanResume(raw); const sentences = extractSentences(resume); const strengths = document.getElementById("strengths"); const gaps = document.getElementById("gaps"); const sentenceReview = document.getElementById("sentenceReview"); strengths.innerHTML = ""; gaps.innerHTML = ""; sentenceReview.innerHTML = ""; let good = 0; let bad = 0; sentences.forEach(s => { const score = scoreSentence(s); if (score >= 3) { good++; strengths.innerHTML += `
  • ${s}
    • `; sentenceReview.innerHTML += `
    ${s}
    `; } else { bad++; gaps.innerHTML += `
  • ${s}
    • `; sentenceReview.innerHTML += `
    ${s}
    `; } }); const finalScore = good + bad === 0 ? 0 : Math.round((good / (good + bad)) * 100); document.getElementById("score").innerText = finalScore; document.getElementById("results").classList.remove("hidden"); document.getElementById("before").innerText = gaps.textContent.slice(0, 600); document.getElementById("after").innerText = "Upgrade to rewrite low-impact sentences automatically."; window.__weakText = gaps.textContent; }); } function rewriteWithAI() { alert("AI rewrite will be enabled after backend is connected (Pro feature)."); }
